\section{End-to-End Example}

% --- Slide 1: Problem Framing ---
\begin{frame}{E2E Example: Inventory Optimization}

\textbf{Context:} A retailer operates dozens of warehouses. A mathematical optimization model allocates inventory across locations weekly.\\[0.5em]

\textbf{The challenge:}
\begin{itemize}
  \item Planners need to \textbf{understand} why inventory distributions changed
  \item They must \textbf{assess} whether the model output is trustworthy given real-world events (port closures, demand surges, supplier delays)
  \item They need to \textbf{decide} whether to apply bias corrections or edge adjustments to the model output
  \item Information is spread across \textbf{databases}, \textbf{codebases}, and \textbf{domain documentation}
\end{itemize}

\vspace{0.5em}
\begin{block}{Two-Step Agentic Approach}
\begin{enumerate}
  \item \textbf{Agentic Chatbot} --- interactive copilot with limited tools; planner drives the analysis
  \item \textbf{Autonomous Agentic Workflow} --- continuously monitors, adapts, and executes; agent drives the process
\end{enumerate}
\end{block}

\vspace{0.3em}
\small\textit{The path to the answer is uncertain, knowledge is spread across heterogeneous sources, and the analysis requires reasoning --- this is when agents add value.}

\end{frame}
